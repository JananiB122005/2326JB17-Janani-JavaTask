class CountThread extends Thread {
    CountThread(String name) {
        super(name);
    }

    public void run() {
        for (int i = 1; i <= 10; i++) {
            System.out.println(getName() + ": " + i);
            try { Thread.sleep(100); } catch (Exception e) {}
        }
    }

    public static void main(String[] args) {
        CountThread t = new CountThread("Counter");
        t.start();
    }
}
